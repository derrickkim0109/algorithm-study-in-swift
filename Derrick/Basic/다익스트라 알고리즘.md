# 다익스트라 알고리즘Dijkstra

```swift 


let number = 6
let INFINITY = 1000_000_000

let a: [[Int]] = [
    [0, 2, 5, 1, INFINITY, INFINITY],
    [2, 0, 3, 2, INFINITY, INFINITY],
    [5, 3, 0, 3, 1, 5],
    [1, 2, 3, 0, 1, INFINITY],
    [INFINITY, INFINITY, 1, 1, 0, 2],
    [INFINITY, INFINITY, 5, INFINITY, 2, 0],
]

// 방문한 노드
var visited = Array(repeating: false, count: 6)

// 최단 거리
var shortestArray: [Int] = []

// 가장 최소 거리를 가지는 정점을 반환한다.

func getShortestIndex() -> Int {
    var min = INFINITY
    var index = 0

    for i in 0..<number {
        if shortestArray[i] < min && !visited[i] {
            min = shortestArray[i]
            index = i
        }
    }

    return index
}

// 다익스트라 수행하는 함수

func dijkstra(start: Int) {
    for i in 0..<number {
        shortestArray[i] = a[start][i]
    }

    visited[start] = true

    for _ in 0..<number - 2 {
        var current = getShortestIndex()
        visited[current] = true

        for j in 0..<number {
            if !visited[j] {
                if shortestArray[current] + a[current][j] < shortestArray[j] {
                    shortestArray[j] = shortestArray[current] + a[current][j]
                }
            }
        }
    }
}

```
